            init: function() {
                // Set language from localStorage or default to 'ru'
                this.currentLang = localStorage.getItem('lang') || 'ru';
                this.updateLanguageButtons();
                this.applyLanguage();
                
                // Add event listeners to language buttons
                document.querySelectorAll('.language-btn').forEach(btn => {
                    btn.addEventListener('click', () => {
                        this.currentLang = btn.dataset.lang;
                        localStorage.setItem('lang', this.currentLang);
                        this.updateLanguageButtons();
                        this.applyLanguage();
                    });
                });
            },
            updateLanguageButtons: function() {
                document.querySelectorAll('.language-btn').forEach(btn => {
                    if (btn.dataset.lang === this.currentLang) {
                        btn.classList.add('active');
                    } else {
                        btn.classList.remove('active');
                    }
                });
            },
            applyLanguage: function() {
                const langData = this.translations[this.currentLang];
                
                // Update all elements with data-i18n attribute
                document.querySelectorAll('[data-i18n]').forEach(el => {
                    const key = el.getAttribute('data-i18n');
                    if (langData[key]) {
                        if (el.tagName === 'INPUT' || el.tagName === 'TEXTAREA') {
                            el.value = langData[key];
                        } else {
                            el.textContent = langData[key];
                        }
                    }
                });
            }
        };
        
        // Initialize language switcher
        languageSwitcher.init();
        
        // Filter commands by category
        document.querySelectorAll('.category-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                // Remove active class from all buttons
                document.querySelectorAll('.category-btn').forEach(b => {
                    b.classList.remove('active');
                });
                
                // Add active class to clicked button
                btn.classList.add('active');
                
                const category = btn.dataset.category;
                const commands = document.querySelectorAll('.command-card');
                
                commands.forEach(cmd => {
                    if (category === 'all' || cmd.dataset.category === category) {
                        cmd.style.display = 'block';
                    } else {
                        cmd.style.display = 'none';
                    }
                });
            });
        });
        
        // Smooth scrolling for anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();
                
                const targetId = this.getAttribute('href');
                const targetElement = document.querySelector(targetId);
                
                if (targetElement) {
                    window.scrollTo({
                        top: targetElement.offsetTop - 80,
                        behavior: 'smooth'
                    });
                }
            });
        });
    </script>
